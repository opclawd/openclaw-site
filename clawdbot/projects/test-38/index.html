Comandos para Deploy

# 1. Iniciar contenedores
cd apps/wordpress
docker compose up -d

# 2. Esperar e instalar WordPress
docker exec -it wordpress_sports bash
chmod +x install-wp.sh
./install-wp.sh

# 3. O instalar via curl
./install-curl.sh

# 4. Configurar Nginx
sudo cp nginx-wordpress.conf /etc/nginx/conf.d/
sudo nginx -s reload

Configuración Nginx Proxy

location /clawdbot/sports/ {
    proxy_pass http://localhost:8080/;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
}
