Consigna

Crear un script Bash que:

  • Busque archivos .log en el directorio logs/
  • Cuente ocurrencias de ERROR y WARNING usando grep
  • Si no hay logs, cree logs/sample.log con datos de ejemplo
  • Guarde el resumen en reports/log_summary.txt
  • Ejecute el script

Script log_analyzer.sh

#!/bin/bash

# Log Analyzer Script
LOG_DIR="logs"
REPORT_FILE="reports/log_summary.txt"
SAMPLE_LOG="logs/sample.log"

# Crear directorios si no existen
mkdir -p "$LOG_DIR" "reports"

# Buscar archivos .log
LOG_FILES=$(find "$LOG_DIR" -name "*.log" -type f 2>/dev/null)

# Si no hay logs, crear sample.log
if [ -z "$LOG_FILES" ]; then
    cat > "$SAMPLE_LOG" << 'EOF'
2024-01-15 10:23:45 INFO Application started
2024-01-15 10:24:12 ERROR Connection timeout
2024-01-15 10:24:15 WARNING Retrying connection
EOF
    LOG_FILES=$(find "$LOG_DIR" -name "*.log" -type f)
fi

# Contadores
TOTAL_FILES=0
TOTAL_ERROR=0
TOTAL_WARNING=0
TOTAL_LINES=0

# Procesar cada archivo
for LOG_FILE in $LOG_FILES; do
    TOTAL_FILES=$((TOTAL_FILES + 1))
    FILE_NAME=$(basename "$LOG_FILE")
    FILE_LINES=$(wc -l < "$LOG_FILE")
    FILE_ERROR=$(grep -c "ERROR" "$LOG_FILE" || echo 0)
    FILE_WARNING=$(grep -c "WARNING" "$LOG_FILE" || echo 0)
    
    TOTAL_LINES=$((TOTAL_LINES + FILE_LINES))
    TOTAL_ERROR=$((TOTAL_ERROR + FILE_ERROR))
    TOTAL_WARNING=$((TOTAL_WARNING + FILE_WARNING))
done

# Guardar reporte
echo "..." > "$REPORT_FILE"
